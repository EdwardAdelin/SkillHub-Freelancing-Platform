@startuml
left to right direction
skinparam packageStyle rectangle

' ACTORS
actor "User" as user
actor "Client" as client
actor "Freelancer" as freelancer
actor "Administrator" as admin

' INHERITANCE
' Client and Freelancer are types of Users
client --|> user
freelancer --|> user

rectangle "SkillHub Platform" {

  ' USER USE CASES (FR1, FR2, FR9)
  usecase "FR1: Register / Login" as FR1
  usecase "FR2: Manage Profile" as FR2
  usecase "FR9: View Status Updates" as FR9
  
  user --> FR1
  user --> FR2
  user --> FR9

  ' CLIENT USE CASES (FR3, FR4, FR8, FR10)
  usecase "FR3: Post Job" as FR3
  usecase "FR4: Manage Posted Jobs" as FR4
  usecase "FR8: Contact Freelancer" as FR8
  usecase "FR10: Hire Freelancer" as FR10
  
  client --> FR3
  client --> FR4
  client --> FR8
  client --> FR10

  ' FREELANCER USE CASES (FR5, FR6, FR7, FR11)
  usecase "FR5: Browse Jobs" as FR5
  usecase "FR6: Submit Proposal" as FR6
  usecase "FR7: Contact Client" as FR7
  usecase "FR11: Complete Job" as FR11
  
  freelancer --> FR5
  freelancer --> FR6
  freelancer --> FR7
  freelancer --> FR11

  ' ADMIN USE CASES (FR12, FR13)
  usecase "FR12: Moderate Platform\n(via Firebase Console)" as FR12
  usecase "FR13: Manage Users\n(via Firebase Console)" as FR13
  
  admin --> FR12
  admin --> FR13
}
@enduml
